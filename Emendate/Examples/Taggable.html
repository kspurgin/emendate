def set_up_tags(data_sets, date_types)
  @data_sets = data_sets.split(';')
  @date_types = date_types.split(';')
  %i[data_sets date_types].each do |tag|
    self.class.define_method(tag){ instance_variable_get("@#{tag}".to_sym) }
  end
end
