def initialize
  @rows = CSV.parse(File.read(Emendate.examples.file_path.call),
    headers: true)
    .map do |row|
    Row.new(row)
  end
end

def retrieve_rows(str, opt)
  rows.select { |row| row.test_string == str }
    .select { |row| row.test_options == opt }
end
