def call
  return(Success(tokens)) unless str.match?(Re)

  result = Emendate::SegmentSets::MixedSet.new
  result << Emendate::DateTypes::Unprocessable.new(
    children: tokens.segments
  )
  result.warnings << 'Unprocessable string'
  Failure(result)
end
