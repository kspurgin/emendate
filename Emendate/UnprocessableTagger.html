def call
  return(Success(tokens)) unless str.match?(Re)

  result = tokens.class.new.copy(tokens)
  result.clear
  result << Emendate::DateTypes::Error.new(
    error_type: :unprocessable,
    sources: tokens
  )
  result.warnings << "Unprocessable string"
  Failure(result)
end
