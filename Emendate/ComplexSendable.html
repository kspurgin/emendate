def conditional_send(meth, arg)
  if arg.is_a?(Array)
    send(meth, *arg)
  elsif arg.is_a?(Hash)
    send(meth, **arg)
  elsif arg.is_a?(Proc)
    real_arg = arg.call
    conditional_send(meth, real_arg)
  else
    send(meth, arg)
  end
end
