def initialize(tokens)
  if tokens.is_a?(String)
    @scanner = StringScanner.new(tokens)
    @tokens = Emendate::SegmentSet.new(string: tokens)
  else
    @tokens = tokens
    @scanner = StringScanner.new(tokens.orig_string)
  end
end
