def call
  while standardizable
    function = determine_standardizer
    break if function.nil?

    pre = result.types.dup
    function.call
    TokenCollapser.call(result).either(
      ->(success) { @result = success },
      ->(failure) { next }
    )
    break if result.types == pre
  end
  Success(result)
end
