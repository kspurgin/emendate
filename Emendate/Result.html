def initialize(pm)
  @pm = pm
  @original_string = pm.orig_string
  @errors = map_errors
  @warnings = pm.warnings
  if pm.state == :failed
    @dates = []
  else
    @dates = pm.tokens.select{ |t| t.date_type? }
      .map do |t|
        Emendate::ParsedDate.new(
        date: t,
        certainty: pm.tokens.certainty,
        orig: original_string
      )
      end
  end
end
